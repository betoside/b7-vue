#42: Executando funções fora do componente

    enviar informações de dentro de um componente para fora dele

    

        
            Corpo da postagem 1
        
        
            Corpo da postagem 2
        

        

            Respondendo a: {{ respostaTitulo }}
            

        


    

    

    div id="v_42">
        postagem titulo="Titulo da postagem 1" v-on:responder="abrirResposta">
            Corpo da postagem 1
        /postagem>
        postagem titulo="Titulo da postagem 2" v-on:responder="abrirResposta">
            Corpo da postagem 2
        /postagem>

        div v-if="responderAberto">
            span>Respondendo a: {{ respostaTitulo }}/span>
            textarea>/textarea>

        /div>

    /div>
    script type="text/javascript">
        let postagem = {
            props:[ 'titulo' ],
            methods: {
                responder: function(){
                    this.$emit('responder', this.titulo);
                }
            },
            template:`
                div>
                    h5>{{ titulo }}/h5>
                    p>slot>/slot>/p>
                    button v-on:click="responder">Responder/button>
                /div>
            `
        }

        let v_42 = new Vue({
            el:'#v_42',
            data:{
                responderAberto:false,
                respostaTitulo:''
            },
            methods: {
                abrirResposta: function(titulo){
                    this.responderAberto = true;
                    this.respostaTitulo = titulo;
                }
            },
            components:{
                postagem
            }
        });
    /script>
